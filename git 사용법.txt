git commit -m "주석쓸 내용" 으로 커밋

git pull origin main : 원격 저장소에서 로컬로 가져옴

git branch 브랜치이름  # 새로운 브랜치 생성
git checkout 브랜치이름  # 새로운 브랜치로 이동
